class MainActivity : AppCompatActivity() {
    lateinit var name: EditText

    lateinit var diaTrabajdos:EditText
    lateinit var rbMaestro:RadioButton
    lateinit var rbOperario:RadioButton
    lateinit var rbPeon:RadioButton

    lateinit var rbEstable:RadioButton
    lateinit var rbContacto:RadioButton

    lateinit var btnCalcular:Button;
    lateinit var btnNuevo:Button

    lateinit var resIngreso:EditText
    lateinit var resBonif:EditText
    lateinit var resTotal:EditText

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        enableEdgeToEdge()
        setContentView(R.layout.activity_main)

        title = "text View Ediext"
        name = findViewById(R.id.edtName)

        diaTrabajdos = findViewById(R.id.edtDiasTrabajados)
        rbMaestro = findViewById(R.id.rbMaestro)
        rbOperario = findViewById(R.id.rbOperario)
        rbPeon = findViewById(R.id.rbPeon)

        rbEstable = findViewById(R.id.rbEstable)
        rbContacto = findViewById(R.id.rbContratado)

        btnCalcular = findViewById(R.id.btnCalcular)
        btnNuevo = findViewById(R.id.btnNuevo)

        resTotal = findViewById(R.id.edtResTotal)
        resBonif = findViewById(R.id.edtresBonificacion)
        resIngreso = findViewById(R.id.edtResIngresos)

        btnCalcular.setOnClickListener ({

            var diasT = diaTrabajdos.text.toString().toFloat()

            var Total = 0f
            var ingresos = 0f
            var bonif = 0f

            when(true){
                rbMaestro.isChecked -> ingresos = diasT * 150f
                rbOperario.isChecked -> ingresos = diasT * 95f
                rbPeon.isChecked -> ingresos = diasT * 75f
                else -> Toast.makeText(applicationContext,"Elija Una Categoria",Toast.LENGTH_SHORT).show()

            }

            if (rbEstable.isChecked){
                bonif = ingresos * 0.20f

            } else if (rbContacto.isChecked){
                bonif = ingresos * 0.10f
            } else {
                Toast.makeText(applicationContext,"Elija una Condicion",Toast.LENGTH_SHORT).show()
            }

            Total = ingresos + bonif

            resIngreso.setText(ingresos.toString())
            resBonif.setText(bonif.toString())
            resTotal.setText(Total.toString())

        })

        btnNuevo.setOnClickListener({

            resTotal.setText("")
            resBonif.setText("")
            resIngreso.setText("")
            diaTrabajdos.setText("")
            name.setText("")
            name.requestFocus()

            rbMaestro.isChecked = false
            rbOperario.isChecked  = false
            rbPeon.isChecked = false

            rbEstable.isChecked = false
            rbContacto.isChecked= false



        })

        ViewCompat.setOnApplyWindowInsetsListener(findViewById(R.id.main)) { v, insets ->
            val systemBars = insets.getInsets(WindowInsetsCompat.Type.systemBars())
            v.setPadding(systemBars.left, systemBars.top, systemBars.right, systemBars.bottom)
            insets
        }
    }
}
